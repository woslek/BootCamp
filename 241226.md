## String Interpolation
name = '철수'

print(f'\'안녕,{name}야\'')


name = '철수'

print('\'안녕, %s'%(name), end='야\'')

## 네모출력
n=5
m=9

print(('*' * n + '\n') * m, end='')

## Dictionary 만들기

name_age ={
    '김중훈': 11
    '전초희': 12
    '지민성': 13
    '김주성': 14
    '남재현': 15
    '권가희': 16
    '최예진': 11
    '하영한': 12
    '안 훈': 13
    '김다영':14
}
for name in name_age:   
    print(f'이름:{name} 나이:{name_age[name]}입니다')
